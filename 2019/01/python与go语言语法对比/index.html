type Person struct {
  name string
  age int
}

func NewPerson(name string, age int) *Person {
  p := &Person{name, age}
  return p
}

func (self *Person) myfunc() {
  fmt.Println(“Hello my name is ” + self.name)
}

p1 := NewPerson(“John”, 36)

fmt.Println(p1.name, p1.age)
